# Phase 2 (v1.1) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship wechat-cc v1.1 with three bundled features — revived `--dangerously` CLI flag for admin-bypass permission model, outbound voice replies via TTS (Qwen default + OpenAI fallback), and a proactive Companion layer with per-project pickable personas (小助手 / 陪伴) plus cron-scheduled Claude-task triggers.

**Architecture:** Phase 1's Agent SDK daemon gains three cohesive additions:
- **Permission-relay polish** (smallest): daemon-level `permissionMode` branches on `--dangerously` flag; Phase 1 relay preserved unchanged for the strict path; trigger eval sessions always bypass (already decided).
- **Voice (outbound only)**: new `src/daemon/tts/` module with `TTSProvider` abstraction; `reply_voice` MCP tool calls provider → uploads audio via existing `ilink` + `media.ts` path; `voice-config.json` stores provider choice + API keys (configured via conversation through `save_voice_config` tool).
- **Companion (proactive)**: new `src/daemon/companion/` module with cron-based scheduler (using `croner`); each trigger spawns an isolated Agent SDK `query()` with persona + profile injected into systemPrompt; Claude either calls `reply` (push) or returns silently (no push); all evals logged to `runs.jsonl` + pushes to `push-log.jsonl`.

Three sub-specs are the source of truth:
- `docs/specs/2026-04-22-permission-relay.md`
- `docs/specs/2026-04-22-voice.md`
- `docs/specs/2026-04-22-companion.md`

**Tech Stack:**
- Phase 1 stack (Bun 1.3+, `@anthropic-ai/claude-agent-sdk@0.2.116`, zod@^4, vitest@^4) — unchanged.
- **New runtime dep**: `croner` (~15KB) for cron parsing + DST-aware scheduling.
- **No new transitive deps**: TTS providers use fetch; YAML front-matter uses a ~20-line custom parser (no `gray-matter`).

**Target release**: v1.1.0-rc.1 after Task 25.

```ts
#!/usr/bin/env bun
/**
 * Phase 2 Spike 5: Agent SDK isolated eval session cold-start + prompt-cache behavior.
 *
 * Goal: measure p50/p95 wall time + cost for 20 sequential fresh query() calls with
 * same systemPrompt + varying user prompt. Target: p50 < 5s per eval.
 */
import { query } from '@anthropic-ai/claude-agent-sdk'

const SYSTEM_PROMPT = `
You are the trigger-evaluation agent for wechat-cc companion.
Persona: assistant.
Rules: if the task asks a question that requires no push, just acknowledge silently.
`.trim()

const TASKS = Array.from({ length: 20 }, (_, i) => `
Task eval #${i + 1}: the current local hour is 14. The last push was 3 hours ago.
Do nothing — just acknowledge. Do NOT call any tool. Respond with the single
word: silent.
`.trim())

const results: { idx: number; durationMs: number; totalCostUsd?: number; numTurns?: number }[] = []

for (let i = 0; i < TASKS.length; i++) {
  const started = Date.now()
  let cost: number | undefined
  let turns: number | undefined
  const q = query({
    prompt: TASKS[i]!,
    options: {
      cwd: process.cwd(),
      permissionMode: 'bypassPermissions',
      systemPrompt: SYSTEM_PROMPT,
    },
  })
  for await (const msg of q) {
    if ((msg as any).type === 'result') {
      const r = msg as any
      cost = r.total_cost_usd
      turns = r.num_turns
    }
  }
  const durationMs = Date.now() - started
  results.push({ idx: i + 1, durationMs, totalCostUsd: cost, numTurns: turns })
  console.log(`[spike5] eval #${i + 1}: ${durationMs}ms, cost=$${cost?.toFixed(4)}, turns=${turns}`)
}

const durations = results.map(r => r.durationMs).sort((a, b) => a - b)
const p50 = durations[Math.floor(durations.length / 2)]
const p95 = durations[Math.floor(durations.length * 0.95)]
const totalCost = results.reduce((s, r) => s + (r.totalCostUsd ?? 0), 0)

console.log('\n=== Spike 5 result ===')
console.log(`p50: ${p50}ms, p95: ${p95}ms`)
console.log(`total cost for 20 evals: $${totalCost.toFixed(4)} (avg $${(totalCost/20).toFixed(4)}/eval)`)

if ((p50 ?? Infinity) <= 5000) {
  console.log('[spike5] PASS ✅ (p50 ≤ 5s)')
  process.exit(0)
} else {
  console.error('[spike5] FAIL ❌ (p50 > 5s — need to pool sessions)')
  process.exit(1)
}
```

```ts
#!/usr/bin/env bun
/**
 * Phase 2 Spike 6: Tool-count impact on model tool-selection quality.
 *
 * Goal: compare Claude's tool-call accuracy at 11 tools (Phase 1 baseline) vs
 * 22 tools (post-v1.1 count) on a synthetic benchmark.
 *
 * Method: register N stub tools (one "real" correct tool + N-1 distractors),
 * ask Claude to perform a specific action that requires the real tool, observe
 * whether Claude selects the correct tool.
 */
import { query, tool, createSdkMcpServer } from '@anthropic-ai/claude-agent-sdk'
import { z } from 'zod'

type TrialResult = { toolCount: number; trial: number; correctToolCalled: boolean; actualTool: string }

async function runTrial(toolCount: number, trial: number): Promise<TrialResult> {
  const tools = [
    tool('send_report', 'Send a daily report to the user via WeChat.', 
      { content: z.string() }, async () => ({ content: [{ type: 'text' as const, text: 'ok' }] })),
    ...Array.from({ length: toolCount - 1 }, (_, i) => 
      tool(`distractor_${i}`, `Distractor tool ${i} — does something unrelated.`,
        {}, async () => ({ content: [{ type: 'text' as const, text: 'ok' }] }))
    ),
  ]
  const cfg = createSdkMcpServer({ name: 'bench', version: '1.0.0', tools })
  
  let actualTool = 'none'
  const q = query({
    prompt: 'Send the daily report to the user: "All systems green."',
    options: {
      cwd: process.cwd(),
      permissionMode: 'bypassPermissions',
      mcpServers: { bench: cfg },
    },
  })
  for await (const msg of q) {
    if ((msg as any).type === 'assistant') {
      const content = (msg as any).message?.content
      if (Array.isArray(content)) {
        for (const b of content) {
          if (b?.type === 'tool_use') actualTool = b.name
        }
      }
    }
  }
  return { toolCount, trial, correctToolCalled: actualTool === 'send_report', actualTool }
}

async function main() {
  const results: TrialResult[] = []
  for (const count of [11, 22]) {
    for (let t = 0; t < 5; t++) {
      const r = await runTrial(count, t + 1)
      console.log(`[spike6] N=${count} trial ${t+1}: selected="${r.actualTool}" correct=${r.correctToolCalled}`)
      results.push(r)
    }
  }
  
  const by = (n: number) => results.filter(r => r.toolCount === n)
  const acc = (arr: TrialResult[]) => arr.filter(r => r.correctToolCalled).length / arr.length
  
  console.log(`\n=== Spike 6 result ===`)
  console.log(`N=11 accuracy: ${(acc(by(11)) * 100).toFixed(0)}%`)
  console.log(`N=22 accuracy: ${(acc(by(22)) * 100).toFixed(0)}%`)
  
  const regression = acc(by(11)) - acc(by(22))
  if (regression <= 0.2) {
    console.log('[spike6] PASS ✅ (regression ≤ 20%)')
  } else {
    console.error(`[spike6] FAIL ❌ (regression ${(regression*100).toFixed(0)}% — split MCP servers)`)
  }
}
main()
```

Change `CliArgs` type and `parseCliArgs` body:

```ts
export type CliArgs =
  | { cmd: 'run'; dangerouslySkipPermissions: boolean }
  | { cmd: 'setup' }
  | { cmd: 'install'; userScope: boolean }
  | { cmd: 'status' }
  | { cmd: 'list' }
  | { cmd: 'help' }

export function parseCliArgs(argv: string[], opts?: { warn?: (m: string) => void }): CliArgs {
  const warn = opts?.warn ?? ((m: string) => console.warn(m))
  const [cmd, ...rest] = argv
  if (!cmd || cmd === '--help' || cmd === '-h' || cmd === 'help') return { cmd: 'help' }
  switch (cmd) {
    case 'run': {
      let dangerouslySkipPermissions = false
      for (const a of rest) {
        if (a === '--dangerously') dangerouslySkipPermissions = true
        else if (a === '--fresh' || a === '--continue' || a === '--channels' || a.startsWith('--mcp-config')) {
          warn(`[wechat-cc] legacy flag ignored: ${a} (v1.0+ daemon doesn't spawn claude directly)`)
        }
      }
      return { cmd: 'run', dangerouslySkipPermissions }
    }
    case 'setup': return { cmd: 'setup' }
    case 'install': return { cmd: 'install', userScope: rest.includes('--user') }
    case 'status': return { cmd: 'status' }
    case 'list': return { cmd: 'list' }
    default: return { cmd: 'help' }
  }
}
```

```ts
// src/daemon/tts/qwen.ts
import type { TTSProvider } from './types'
import { Buffer } from 'node:buffer'

export interface QwenProviderOptions {
  apiKey: string
  /** Defaults to 'qwen3-tts-flash' */
  model?: string
  /** Defaults to the public DashScope endpoint */
  baseUrl?: string
}

// VERIFY during impl: DashScope TTS endpoint path may need adjustment.
// At spec-writing time the documented path is /api/v1/services/audio/tts/
// which returns audio directly (not JSON). If the deployed API differs, adjust here.
const DEFAULT_ENDPOINT = 'https://dashscope.aliyuncs.com/api/v1/services/audio/tts/'

export function makeQwenProvider(opts: QwenProviderOptions): TTSProvider {
  const endpoint = opts.baseUrl ?? DEFAULT_ENDPOINT
  const model = opts.model ?? 'qwen3-tts-flash'

  async function synth(text: string, voice: string) {
    const res = await fetch(endpoint, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${opts.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model,
        input: { text },
        parameters: { voice },
      }),
    })
    if (!res.ok) {
      const body = await res.text().catch(() => '')
      throw new Error(`Qwen TTS ${res.status}: ${body.slice(0, 200)}`)
    }
    const buf = Buffer.from(await res.arrayBuffer())
    const mimeType = res.headers.get('content-type') ?? 'audio/mpeg'
    return { audio: buf, mimeType }
  }

  async function test(): Promise<{ ok: true } | { ok: false; reason: string; detail?: string }> {
    try {
      await synth('测试', 'Cherry')
      return { ok: true }
    } catch (err) {
      const detail = err instanceof Error ? err.message : String(err)
      const status = /\b(\d{3})\b/.exec(detail)?.[1]
      const reason = status === '401' ? 'unauthorized (check api key)'
        : status === '429' ? 'rate limited'
        : /^5\d\d/.test(status ?? '') ? 'qwen service error'
        : 'unknown'
      return { ok: false, reason, detail }
    }
  }

  return { name: 'qwen', synth, test }
}
```

```ts
// src/daemon/tts/voice-config.ts
import { readFileSync, writeFileSync, existsSync, chmodSync, renameSync } from 'node:fs'
import { join } from 'node:path'

export interface VoiceConfig {
  provider: 'qwen' | 'openai'
  api_key: string
  default_voice?: string
  openai_api_key?: string
  saved_at: string
}

function configPath(stateDir: string): string {
  return join(stateDir, 'voice-config.json')
}

export function loadVoiceConfig(stateDir: string): VoiceConfig | null {
  const p = configPath(stateDir)
  if (!existsSync(p)) return null
  try {
    const parsed = JSON.parse(readFileSync(p, 'utf8')) as unknown
    if (!parsed || typeof parsed !== 'object') return null
    const c = parsed as VoiceConfig
    if (!['qwen', 'openai'].includes(c.provider)) return null
    if (typeof c.api_key !== 'string' || c.api_key.length < 3) return null
    return c
  } catch { return null }
}

export async function saveVoiceConfig(stateDir: string, cfg: VoiceConfig): Promise<void> {
  const p = configPath(stateDir)
  const tmp = `${p}.tmp-${process.pid}-${Date.now()}`
  writeFileSync(tmp, JSON.stringify(cfg, null, 2), 'utf8')
  try { chmodSync(tmp, 0o600) } catch { /* Windows may throw; best-effort */ }
  renameSync(tmp, p)
}
```

```ts
import { join } from 'node:path'
import { personasDir, personaPath } from './paths'

export interface PersonaFrontmatter {
  name: string
  display_name: string
  min_push_gap_minutes: number
  quiet_hours_local: string
}

export interface Persona {
  frontmatter: PersonaFrontmatter
  body: string
  sourcePath: string
}

const FRONT_MATTER_RE = /^---\r?\n([\s\S]*?)\r?\n---\r?\n([\s\S]*)$/

export function parsePersonaFile(content: string, sourcePath: string): Persona | null {
  const m = FRONT_MATTER_RE.exec(content)
  if (!m) return null
  const rawFm = m[1]!
  const body = m[2]!.trim()
  const fm: Record<string, unknown> = {}
  for (const line of rawFm.split(/\r?\n/)) {
    const eq = line.indexOf(':')
    if (eq < 0) continue
    const k = line.slice(0, eq).trim()
    let v: string = line.slice(eq + 1).trim()
    if (v.startsWith('"') && v.endsWith('"')) v = v.slice(1, -1)
    if (!k) continue
    fm[k] = v
  }
  try {
    const name = String(fm['name'] ?? '')
    const display_name = String(fm['display_name'] ?? '')
    if (!name || !display_name) return null
    const gap = Number(fm['min_push_gap_minutes'] ?? 10)
    if (!Number.isFinite(gap) || gap < 0) return null
    const quiet = String(fm['quiet_hours_local'] ?? '')
    return {
      frontmatter: { name, display_name, min_push_gap_minutes: gap, quiet_hours_local: quiet },
      body,
      sourcePath,
    }
  } catch {
    return null
  }
}

export function listPersonas(stateDir: string): Persona[] {
  const dir = personasDir(stateDir)
  if (!existsSync(dir)) return []
  const out: Persona[] = []
  for (const f of readdirSync(dir)) {
    if (!f.endsWith('.md')) continue
    try {
      const content = readFileSync(join(dir, f), 'utf8')
      const p = parsePersonaFile(content, join(dir, f))
      if (p) out.push(p)
    } catch { /* skip unreadable */ }
  }
  return out
}

export function loadPersona(stateDir: string, name: string): Persona | null {
  const p = personaPath(stateDir, name)
  if (!existsSync(p)) return null
  try {
    return parsePersonaFile(readFileSync(p, 'utf8'), p)
  } catch { return null }
}
```

```ts
import { appendFileSync, existsSync, statSync, renameSync, mkdirSync } from 'node:fs'
import { companionDir, runsPath, pushLogPath } from './paths'

export interface RunEntry {
  ts: string
  trigger_id: string
  persona: string
  duration_ms: number
  pushed: boolean
  reason: string
  tool_uses_count: number
  cost_usd: number
  error_message?: string
}

export interface PushEntry {
  ts: string
  trigger_id: string
  persona: string
  message: string
  chat_id: string
  delivery_status: 'ok' | 'failed'
  error?: string
}

const ROTATE_BYTES = 10 * 1024 * 1024  // 10 MB

function makeLogger<T>(path: string) {
  function ensureDir() {
    const d = path.replace(/[/\\][^/\\]+$/, '')
    if (!existsSync(d)) mkdirSync(d, { recursive: true })
  }
  return {
    append(entry: T): void {
      ensureDir()
      appendFileSync(path, JSON.stringify(entry) + '\n', 'utf8')
    },
    rotate(): void {
      if (!existsSync(path)) return
      try {
        const size = statSync(path).size
        if (size >= ROTATE_BYTES) renameSync(path, `${path}.1`)
      } catch { /* best effort */ }
    },
  }
}

export function makeRunsLogger(stateDir: string) {
  return makeLogger<RunEntry>(runsPath(stateDir))
}

export function makePushLogger(stateDir: string) {
  return makeLogger<PushEntry>(pushLogPath(stateDir))
}
```

```ts
// src/daemon/companion/scheduler.ts
import { Cron } from 'croner'
import type { CompanionConfig, Trigger } from './config'
import type { Persona } from './persona'
import type { RunEntry, PushEntry } from './logs'

export interface EvalContext {
  recent_pushes: PushEntry[]
  recent_runs: RunEntry[]
  profile: string
  persona: Persona
  chat_id: string
}

export interface EvalResult {
  pushed: boolean
  message?: string
  cost_usd: number
  tool_uses_count: number
  duration_ms: number
  error_message?: string
}

export interface SchedulerDeps {
  loadConfig: () => CompanionConfig
  runs: { append: (e: RunEntry) => void; rotate: () => void }
  pushes: { append: (e: PushEntry) => void; rotate: () => void }
  evalTrigger: (trigger: Trigger, ctx: { cfg: CompanionConfig }) => Promise<EvalResult>
  now: () => Date
}

const TICK_MS = 60_000

export function startScheduler(deps: SchedulerDeps, log: (tag: string, line: string) => void): () => Promise<void> {
  let stopped = false
  const timer = setInterval(async () => {
    if (stopped) return
    try { await tick(deps, log) } catch (err) { log('SCHED', `tick error: ${err}`) }
    // rotation on every tick is cheap
    deps.runs.rotate()
    deps.pushes.rotate()
  }, TICK_MS)

  return async () => {
    stopped = true
    clearInterval(timer)
  }
}

async function tick(deps: SchedulerDeps, log: (tag: string, line: string) => void): Promise<void> {
  const cfg = deps.loadConfig()
  if (!cfg.enabled) return
  if (cfg.snooze_until && new Date(cfg.snooze_until).getTime() > deps.now().getTime()) {
    log('SCHED', `snoozed until ${cfg.snooze_until}`)
    return
  }
  const currentMinute = deps.now()
  for (const trigger of cfg.triggers) {
    try {
      if (trigger.paused_until && new Date(trigger.paused_until).getTime() > currentMinute.getTime()) continue
      const cron = new Cron(trigger.schedule, { timezone: cfg.timezone, paused: true }, () => {})
      if (!cron.match(currentMinute)) continue

      const started = Date.now()
      const result = await deps.evalTrigger(trigger, { cfg })
      const duration_ms = Date.now() - started

      deps.runs.append({
        ts: currentMinute.toISOString(),
        trigger_id: trigger.id,
        persona: cfg.per_project_persona[trigger.project] ?? cfg.per_project_persona['_default'] ?? 'assistant',
        duration_ms,
        pushed: result.pushed,
        reason: result.pushed ? 'ok' : 'no_push',
        tool_uses_count: result.tool_uses_count,
        cost_usd: result.cost_usd,
        error_message: result.error_message,
      })

      if (result.pushed && result.message) {
        deps.pushes.append({
          ts: currentMinute.toISOString(),
          trigger_id: trigger.id,
          persona: cfg.per_project_persona[trigger.project] ?? cfg.per_project_persona['_default'] ?? 'assistant',
          message: result.message,
          chat_id: cfg.default_chat_id ?? '',
          delivery_status: 'ok',
        })
      }
    } catch (err) {
      log('SCHED', `trigger ${trigger.id} error: ${err}`)
    }
  }
}
```

```ts
export function makeEvalTrigger(deps: EvalSessionDeps): (trigger: Trigger, ctx: EvalContext) => Promise<EvalResult> {
  return async (trigger, ctx) => {
    const systemPrompt = [ctx.profile, '---', ctx.persona.body].join('\n\n')
    const base = deps.sdkOptionsBase()
    const options: Options = {
      ...base,
      permissionMode: 'bypassPermissions',
      systemPrompt,
      settingSources: ['user', 'project', 'local'],
    }
    delete (options as any).canUseTool

    const prompt = buildTaskPrompt(trigger, ctx)

    const started = Date.now()
    let pushed = false
    let message: string | undefined
    let toolUses = 0
    let cost = 0
    let duration = 0

    try {
      const q = query({ prompt, options })
      for await (const msg of q as AsyncIterable<SDKMessage>) {
        const m = msg as any
        if (m.type === 'assistant') {
          const content = m.message?.content
          if (Array.isArray(content)) {
            for (const b of content) {
              if (b?.type === 'tool_use') {
                toolUses++
                if (b.name === 'reply') {
                  pushed = true
                  const text = b.input?.text
                  if (typeof text === 'string') message = text
                }
              }
            }
          }
        } else if (m.type === 'result') {
          cost = m.total_cost_usd ?? 0
          duration = m.duration_ms ?? (Date.now() - started)
        }
      }
    } catch (err) {
      const detail = err instanceof Error ? err.message : String(err)
      deps.log('EVAL', `trigger ${trigger.id} error: ${detail}`)
      return {
        pushed: false, cost_usd: 0, tool_uses_count: 0,
        duration_ms: Date.now() - started,
        error_message: detail,
      }
    }

    return {
      pushed, message,
      cost_usd: cost,
      tool_uses_count: toolUses,
      duration_ms: duration || (Date.now() - started),
    }
  }
}
```
